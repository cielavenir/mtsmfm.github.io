Rspec + Guard + Spork + Notify\
\
Gemfile

~~~~ {.syntax-highlight}
group :test, :development do
  gem 'rspec-rails'
  gem 'spork'
  gem 'guard-spork'
  gem 'guard-rspec'
  gem 'rb-inotify', :require => false
  gem 'libnotify'
  gem 'factory_girl_rails'
end
~~~~

\

Rspecなら

~~~~ {.syntax-highlight}
rails g rspec:install
~~~~

\

Guardなら

~~~~ {.syntax-highlight}
guard init spork
guard init rspec
~~~~

\

Spork.each_run do
